import numpy as np
import matplotlib.pyplot as plt
from sklearn import svm

# Define the mesh grid for plotting decision boundaries
h = 1  # step size in the mesh
x_min, x_max = X['age'].min() - 1, X['age'].max() + 1
y_min, y_max = X['job'].min() - 1, X['job'].max() + 1
xx, yy = np.meshgrid(np.arange(x_min, x_max, h),
                     np.arange(y_min, y_max, h))

# Create subplots
fig, sub = plt.subplots(2, 2, figsize=(10, 10))

# titles for the plots
titles = ['SVC with linear kernel',
          'LinearSVC (linear kernel)',
          'SVC with RBF kernel',
          'SVC with polynomial (degree 3) kernel']

models = [svc, lin_svc, rbf_svc, poly_svc]

for clf, title, ax in zip(models, titles, sub.flatten()):
    # Plot decision boundary
    Z = clf.predict(np.c_[xx.ravel(), yy.ravel()])
    Z = Z.reshape(xx.shape)
    ax.contourf(xx, yy, Z, cmap=plt.cm.coolwarm, alpha=0.8)

    # Plot the training points
    ax.scatter(X['age'], X['job'], c=y, cmap=plt.cm.coolwarm, s=20, edgecolors='k')
    ax.set_xlim(xx.min(), xx.max())
    ax.set_ylim(yy.min(), yy.max())
    ax.set_xlabel('age')
    ax.set_ylabel('job')
    ax.set_xticks(())
    ax.set_yticks(())
    ax.set_title(title)

plt.show()

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn import svm
from sklearn.model_selection import train_test_split
from sklearn import datasets
from sklearn.metrics import accuracy_score

x_train,x_test,y_train,y_test=train_test_split(X,y,test_size=0.30,random_state=0)
#SVC with linear function
C=1
svc=svm.SVC(kernel='linear',C=C).fit(X,y) #define and fit the model
#linear svc
lin_svc=svm.LinearSVC(C=C).fit(X,y)
#svc with RBF kernel
rbf_svc=svm.SVC(kernel='rbf',gamma=0.7,C=C).fit(X,y)
#svc with polynomial kernel for degree>1
poly_svc=svm.SVC(kernel='poly',degree=3,C=C).fit(X,y)
for clf in [svc,lin_svc,rbf_svc,poly_svc]:
    y_pred=clf.predict(x_test)
    accuracy=accuracy_score(y_test,y_pred)*100
    print(f"Accuracy for {clf} is:{accuracy}")

  • The accuracy of 4 kernels are quite similar, maybe that's the reason why 4 SVM Graphs look like the same in data distribution.
